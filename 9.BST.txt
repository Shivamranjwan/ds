class Node:
    def __init__(self, data):
        self.data = data
        self.left = self.right = None

def insert(root, value):
    if not root: return Node(value)
    if value < root.data: root.left = insert(root.left, value)
    elif value > root.data: root.right = insert(root.right, value)
    return root

def search(root, value):
    if not root: return False
    if root.data == value: return True
    return search(root.left, value) if value < root.data else search(root.right, value)

def inorder(root):
    if root:
        inorder(root.left)
        print(root.data, end=' ')
        inorder(root.right)

def delete(root, value):
    if not root: return root
    if value < root.data: root.left = delete(root.left, value)
    elif value > root.data: root.right = delete(root.right, value)
    else:
        if not root.left: return root.right
        if not root.right: return root.left
        temp = root.right
        while temp.left: temp = temp.left
        root.data = temp.data
        root.right = delete(root.right, temp.data)
    return root

root = None
while True:
    print("\n1.Insert 2.Search 3.Delete 4.Display 5.Exit")
    choice = int(input("Choice: "))
    if choice == 1:
        root = insert(root, int(input("Value: ")))
    elif choice == 2:
        print("Found" if search(root, int(input("Value: "))) else "Not Found")
    elif choice == 3:
        root = delete(root, int(input("Value: ")))
    elif choice == 4:
        inorder(root); print()
    else:
        break
